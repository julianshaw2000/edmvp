# Form SD Support Package — Document Generation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Generate a Form SD Support Package as a single PDF (via QuestPDF) + source JSON, stored to R2 with SHA-256 tamper evidence. Combines applicability determination, supply chain description, due diligence documentation, and risk assessment into one downloadable package.

**Architecture:** New MediatR command `GenerateFormSdPackage` gathers data from the 4 backend generators (Plan A), composes a QuestPDF document, uploads to R2, records metadata in `FormSdPackageEntity`, and updates the filing cycle status to `PACKAGE_READY`. Follows the existing `GeneratePassport.cs` / `GenerateDossier.cs` pattern exactly.

**Tech Stack:** .NET 10, QuestPDF, MediatR, R2 (IFileStorageService), SHA-256.

**Depends on:** Plan A (backend compliance engine) must be complete.

```csharp
using QuestPDF.Fluent;
using QuestPDF.Helpers;
using QuestPDF.Infrastructure;

namespace Tungsten.Api.Features.FormSd.Templates;

public class FormSdPackageTemplate(FormSdPackageData data) : IDocument
{
    public DocumentMetadata GetMetadata() => new()
    {
        Title = $"Form SD Support Package — {data.TenantName} — {data.ReportingYear}",
        Author = "auditraks",
    };

    public void Compose(IDocumentContainer container)
    {
        container.Page(page =>
        {
            page.Size(PageSizes.A4);
            page.Margin(40);
            page.DefaultTextStyle(x => x.FontSize(10).FontFamily("Helvetica"));

            page.Header().Element(ComposeHeader);
            page.Content().Element(ComposeContent);
            page.Footer().AlignCenter().Text(x =>
            {
                x.Span("Page ");
                x.CurrentPageNumber();
                x.Span(" of ");
                x.TotalPages();
                x.Span($" — Generated {data.GeneratedAt:yyyy-MM-dd HH:mm} UTC");
            });
        });
    }

    private void ComposeHeader(IContainer container)
    {
        container.Column(col =>
        {
            col.Item().Text($"FORM SD SUPPORT PACKAGE").Bold().FontSize(16);
            col.Item().Text($"{data.TenantName} — Reporting Year {data.ReportingYear}").FontSize(12).FontColor(Colors.Grey.Darken1);
            col.Item().PaddingTop(5).LineHorizontal(1).LineColor(Colors.Grey.Lighten2);
            col.Item().PaddingTop(5).Row(row =>
            {
                row.RelativeItem().Text($"Generated by: {data.GeneratedBy}").FontSize(8);
                row.RelativeItem().AlignRight().Text($"Rule Set: {data.RuleSetVersion} | Platform: {data.PlatformVersion}").FontSize(8);
            });
            col.Item().PaddingBottom(10);
        });
    }

    private void ComposeContent(IContainer container)
    {
        container.Column(col =>
        {
            // Section 1: Applicability
            col.Item().Text("1. 3TG APPLICABILITY DETERMINATION").Bold().FontSize(13);
            col.Item().PaddingTop(5);
            foreach (var a in data.Applicability)
            {
                col.Item().Row(row =>
                {
                    row.ConstantItem(120).Text(a.BatchNumber).SemiBold();
                    row.ConstantItem(100).Text(a.MineralType);
                    row.ConstantItem(50).Text(a.OriginCountry);
                    row.RelativeItem().Text(a.Status).Bold()
                        .FontColor(a.Status == "IN_SCOPE" ? Colors.Orange.Darken2 :
                                   a.Status == "OUT_OF_SCOPE" ? Colors.Green.Darken2 : Colors.Red.Darken2);
                });
            }

            col.Item().PaddingVertical(15).LineHorizontal(1).LineColor(Colors.Grey.Lighten2);

            // Section 2: Supply Chain Descriptions
            col.Item().Text("2. SUPPLY CHAIN DESCRIPTIONS").Bold().FontSize(13);
            col.Item().PaddingTop(5);
            foreach (var sc in data.SupplyChains)
            {
                col.Item().PaddingBottom(5).Column(inner =>
                {
                    inner.Item().Text($"{sc.BatchNumber} ({sc.EventCount} events, {sc.GapCount} gaps)").SemiBold();
                    inner.Item().Text(sc.NarrativeText).FontSize(9);
                });
            }

            col.Item().PaddingVertical(15).LineHorizontal(1).LineColor(Colors.Grey.Lighten2);

            // Section 3: Due Diligence
            col.Item().Text("3. DUE DILIGENCE PROCESS DOCUMENTATION").Bold().FontSize(13);
            col.Item().PaddingTop(5);
            foreach (var dd in data.DueDiligence)
            {
                col.Item().PaddingBottom(5).Column(inner =>
                {
                    inner.Item().Text($"{dd.BatchNumber} — {dd.FlagCount} flag(s), OECD DDG {dd.OecdVersion}").SemiBold();
                    inner.Item().Text(dd.SummaryText).FontSize(9);
                    if (dd.SmelterNames.Count > 0)
                        inner.Item().Text($"Smelters: {string.Join(", ", dd.SmelterNames)}").FontSize(9).Italic();
                });
            }

            col.Item().PaddingVertical(15).LineHorizontal(1).LineColor(Colors.Grey.Lighten2);

            // Section 4: Risk Assessment
            col.Item().Text("4. RISK ASSESSMENT SUMMARY").Bold().FontSize(13);
            col.Item().PaddingTop(5);
            foreach (var ra in data.RiskAssessments)
            {
                col.Item().PaddingBottom(5).Column(inner =>
                {
                    inner.Item().Text($"{ra.BatchNumber} — Overall: {ra.OverallRating}").SemiBold()
                        .FontColor(ra.OverallRating == "HIGH" ? Colors.Red.Darken2 :
                                   ra.OverallRating == "MEDIUM" ? Colors.Orange.Darken2 : Colors.Green.Darken2);
                    foreach (var cat in ra.Categories)
                        inner.Item().Text($"  {cat.Category}: {cat.Rating} — {cat.Detail}").FontSize(9);
                });
            }
        });
    }
}
```

```csharp
using System.Security.Cryptography;
using System.Text.Json;
using MediatR;
using Microsoft.EntityFrameworkCore;
using QuestPDF.Fluent;
using Tungsten.Api.Common;
using Tungsten.Api.Common.Audit;
using Tungsten.Api.Common.Auth;
using Tungsten.Api.Common.Services;
using Tungsten.Api.Features.FormSd.Templates;
using Tungsten.Api.Infrastructure.Persistence;
using Tungsten.Api.Infrastructure.Persistence.Entities;

namespace Tungsten.Api.Features.FormSd;

public static class GenerateFormSdPackage
{
    public record Command(int ReportingYear) : IRequest<Result<Response>>, IAuditable
    {
        public string AuditAction => "GenerateFormSdPackage";
        public string EntityType => "FormSdPackage";
    }

    public record Response(Guid Id, string DownloadUrl, DateTime GeneratedAt);

    public class Handler(
        AppDbContext db,
        ICurrentUserService currentUser,
        IFileStorageService storage,
        IMediator mediator) : IRequestHandler<Command, Result<Response>>
    {
        private const string PlatformVersion = "2.0.0";

        public async Task<Result<Response>> Handle(Command cmd, CancellationToken ct)
        {
            var user = await db.Users.AsNoTracking()
                .Include(u => u.Tenant)
                .FirstOrDefaultAsync(u => u.IdentityUserId == currentUser.IdentityUserId && u.IsActive, ct);
            if (user is null)
                return Result<Response>.Failure("User not found");

            var tenantId = user.TenantId;

            // Get all batches for this tenant in the reporting year
            var yearStart = new DateTime(cmd.ReportingYear, 1, 1, 0, 0, 0, DateTimeKind.Utc);
            var yearEnd = new DateTime(cmd.ReportingYear + 1, 1, 1, 0, 0, 0, DateTimeKind.Utc);

            var batches = await db.Batches.AsNoTracking()
                .Where(b => b.TenantId == tenantId && b.CreatedAt >= yearStart && b.CreatedAt < yearEnd)
                .ToListAsync(ct);

            if (batches.Count == 0)
                return Result<Response>.Failure("No batches found for the reporting year");

            // Gather data from each generator
            var applicability = new List<BatchApplicability>();
            var supplyChains = new List<BatchSupplyChain>();
            var dueDiligence = new List<BatchDueDiligence>();
            var riskAssessments = new List<BatchRiskAssessment>();

            foreach (var batch in batches)
            {
                // Applicability
                var appResult = await ApplicabilityEngine.EvaluateAsync(db, batch.Id, tenantId, ct);
                applicability.Add(new BatchApplicability(
                    batch.BatchNumber, batch.MineralType, batch.OriginCountry,
                    appResult.Status, appResult.Reasoning?.ToString() ?? ""));

                // Supply chain
                var scResult = await mediator.Send(new GenerateSupplyChainDescription.Query(batch.Id), ct);
                if (scResult.IsSuccess)
                {
                    supplyChains.Add(new BatchSupplyChain(
                        batch.BatchNumber, scResult.Value.NarrativeText,
                        scResult.Value.Chain.Count, scResult.Value.Gaps.Count));
                }

                // Due diligence
                var ddResult = await mediator.Send(new GenerateDueDiligenceSummary.Query(batch.Id), ct);
                if (ddResult.IsSuccess)
                {
                    dueDiligence.Add(new BatchDueDiligence(
                        batch.BatchNumber, ddResult.Value.RiskFlags.Count,
                        ddResult.Value.OecdDdgVersion, ddResult.Value.SummaryText,
                        ddResult.Value.Smelters.Select(s => s.SmelterName).ToList()));
                }

                // Risk assessment
                var raResult = await mediator.Send(new GenerateRiskAssessment.Query(batch.Id), ct);
                if (raResult.IsSuccess)
                {
                    riskAssessments.Add(new BatchRiskAssessment(
                        batch.BatchNumber, raResult.Value.OverallRating,
                        raResult.Value.Categories.Select(c =>
                            new RiskCategoryItem(c.Category, c.Rating, c.Detail)).ToList()));
                }
            }

            var packageData = new FormSdPackageData(
                user.Tenant.Name, cmd.ReportingYear, DateTime.UtcNow,
                user.DisplayName, "1.0.0", PlatformVersion,
                applicability, supplyChains, dueDiligence, riskAssessments);

            // Generate PDF
            var template = new FormSdPackageTemplate(packageData);
            using var pdfStream = new MemoryStream();
            template.GeneratePdf(pdfStream);
            pdfStream.Position = 0;

            // SHA-256 hash
            var hash = Convert.ToHexStringLower(SHA256.HashData(pdfStream.ToArray()));
            pdfStream.Position = 0;

            // Store to R2
            var storageKey = $"{tenantId}/form-sd/{cmd.ReportingYear}/{Guid.NewGuid()}.pdf";
            await storage.UploadAsync(storageKey, pdfStream, "application/pdf", ct);

            // Source JSON
            var sourceJson = JsonSerializer.Serialize(packageData);

            // Save metadata
            var package = new FormSdPackageEntity
            {
                Id = Guid.NewGuid(),
                TenantId = tenantId,
                ReportingYear = cmd.ReportingYear,
                StorageKey = storageKey,
                Sha256Hash = hash,
                RuleSetVersion = "1.0.0",
                PlatformVersion = PlatformVersion,
                GeneratedBy = user.Id,
                SourceJson = sourceJson,
                GeneratedAt = DateTime.UtcNow,
            };
            db.FormSdPackages.Add(package);

            // Update filing cycle status
            var cycle = await db.FormSdFilingCycles
                .FirstOrDefaultAsync(c => c.TenantId == tenantId && c.ReportingYear == cmd.ReportingYear, ct);
            if (cycle is not null && cycle.Status is "NOT_STARTED" or "IN_PROGRESS")
            {
                cycle.Status = "PACKAGE_READY";
                cycle.UpdatedAt = DateTime.UtcNow;
            }

            // Notification
            db.Notifications.Add(new NotificationEntity
            {
                Id = Guid.NewGuid(),
                TenantId = tenantId,
                UserId = user.Id,
                Type = "DOCUMENT_GENERATED",
                Title = "Form SD Support Package ready",
                Message = $"Form SD Support Package for reporting year {cmd.ReportingYear} is ready for download.",
                ReferenceId = package.Id,
                CreatedAt = DateTime.UtcNow,
            });

            await db.SaveChangesAsync(ct);

            var downloadUrl = storage.GetDownloadUrl(storageKey);
            return Result<Response>.Success(new Response(package.Id, downloadUrl, package.GeneratedAt));
        }
    }
}
```

The generator:
1. Gathers all batches for the reporting year
2. Runs applicability, supply chain, due diligence, and risk assessment for each
3. Renders a 4-section PDF via QuestPDF
4. Uploads to R2 at `{tenantId}/form-sd/{year}/{guid}.pdf`
5. Records SHA-256 hash for tamper evidence
6. Updates filing cycle to `PACKAGE_READY`
7. Sends notification
